function getGraphData(idsOfInterest) {
const filteredLinks = mc1.links.filter(link => {
return idsOfInterest.includes(link.target) || idsOfInterest.includes(link.source);
});

const uniqueNodes = Array.from(new Set(filteredLinks.flatMap(link => [link.source, link.target])))
.map(id => mc1.nodes.find(node => node.id === id));

return { nodes: uniqueNodes, links: filteredLinks };
}
